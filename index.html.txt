        function closeProductModal() {
            document.getElementById('productModal').classList.add('hidden');

        function addToCart(productId) {
            const product = premiumProducts.find(p => p.id === productId);
            if (!product) return;

            const existingItem = cart.find(item => item.id === productId);
            if (existingItem) {
                existingItem.quantity += 1;
            } else {
                cart.push({ ...product, quantity: 1 });
            }

            updateCartUI();
            trackAddToCart(product.name, product.code, product.price);
        }

        function removeFromCart(productId) {
            cart = cart.filter(item => item.id !== productId);
            updateCartUI();
        }

        function updateCartUI() {
            const cartCount = document.getElementById('cartCount');
            const totalItems = cart.reduce((sum, item) => sum + item.quantity, 0);
            cartCount.textContent = totalItems;

            const cartItems = document.getElementById('cartItems');
            if (cart.length === 0) {
                cartItems.innerHTML = '<p class="text-gray-500 text-center inter">Sepetiniz boş</p>';
            } else {
                cartItems.innerHTML = cart.map(item => `
                    <div class="flex justify-between items-center py-2 border-b border-gray-100">
                        <div class="flex-1">
                            <h4 class="inter font-medium text-gray-900">${item.name}</h4>
                            <p class="text-sm text-gray-500">Adet: ${item.quantity}</p>
                        </div>
                        <div class="flex items-center gap-2">
                            <span class="inter font-semibold text-amber-700">${item.price}</span>
                            <button onclick="removeFromCart(${item.id})" class="text-red-500 hover:text-red-700">
                                <svg class="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                    <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16"></path>
                                </svg>
                            </button>
                        </div>
                    </div>
                `).join('');
            }
        }

                    'item_id': productCode,
                    'item_name': productName,
                    'category': 'lingerie',
                    'price': parseFloat(price.replace('₺', ''))
                }]
            });
        }

                'value': parseFloat(price.replace('₺', '')),
                'items': [{
                    'item_id': productCode,
                    'item_name': productName,
                    'category': 'lingerie',
                    'price': parseFloat(price.replace('₺', ''))
                }]
            });
        }
